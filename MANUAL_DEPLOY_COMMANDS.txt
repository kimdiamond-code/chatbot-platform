# Manual Deployment Commands

## Run these commands one at a time in PowerShell:

# 1. Navigate to project directory
cd "C:\Users\kdiamond\OneDrive - True Citrus\Directory\Chatbot\Chatbot set up\Chatbot-platform_files\chatbot-platform"

# 2. Check current status
git status

# 3. Add all changes
git add .

# 4. Commit changes
git commit -m "Fix: Bot Builder complete fix with diagnostic test"

# 5. Push to GitHub (this triggers Vercel deployment)
git push origin main

# 6. Wait 2-3 minutes for Vercel to deploy automatically

# 7. Check deployment status
vercel ls

# 8. (Optional) Force manual deploy if GitHub webhook doesn't trigger
vercel --prod
